// ===== STATE =====
var vocab = [];
var timePerQuestion = 10;
var gameState = null;
var timerInterval = null;

// ===== SAFE STORAGE =====
function storageGet(key) {
  try { return localStorage.getItem(key); } catch(e) { return null; }
}
function storageSet(key, val) {
  try { localStorage.setItem(key, val); } catch(e) { /* ignore */ }
}

// ===== INIT =====
function init() {
  try {
    var saved = storageGet("legion_vocab");
    if (saved) {
      vocab = JSON.parse(saved);
    } else {
      vocab = DEFAULT_VOCAB.slice();
    }
  } catch(e) {
    vocab = DEFAULT_VOCAB.slice();
  }
  try {
    var savedDiff = storageGet("legion_difficulty");
    if (savedDiff) {
      timePerQuestion = parseInt(savedDiff, 10);
      var btns = document.querySelectorAll(".diff-btn");
      for (var i = 0; i < btns.length; i++) {
        if (btns[i].getAttribute("data-time") === savedDiff) {
          btns[i].classList.add("active");
        } else {
          btns[i].classList.remove("active");
        }
      }
    }
  } catch(e) { /* ignore */ }
}

function saveVocab() {
  storageSet("legion_vocab", JSON.stringify(vocab));
}

// ===== SCREENS =====
function showScreen(id) {
  var screens = document.querySelectorAll(".screen");
  for (var i = 0; i < screens.length; i++) {
    screens[i].classList.remove("active");
  }
  document.getElementById(id).classList.add("active");
}

function setDifficulty(btn) {
  var btns = document.querySelectorAll(".diff-btn");
  for (var i = 0; i < btns.length; i++) { btns[i].classList.remove("active"); }
  btn.classList.add("active");
  timePerQuestion = parseInt(btn.getAttribute("data-time"), 10);
  storageSet("legion_difficulty", String(timePerQuestion));
}

// ===== GAME =====
function shuffle(arr) {
  var a = arr.slice();
  for (var i = a.length - 1; i > 0; i--) {
    var j = Math.floor(Math.random() * (i + 1));
    var t = a[i]; a[i] = a[j]; a[j] = t;
  }
  return a;
}

function startGame() {
  if (vocab.length < 8) {
    showToast("Mindestens 8 Vokabeln n\u00f6tig!");
    return;
  }

  var shuffled = shuffle(vocab);
  var selected = shuffled.slice(0, Math.min(20, shuffled.length));

  gameState = {
    questions: selected,
    currentIndex: 0,
    score: 0,
    streak: 0,
    maxStreak: 0,
    correct: 0,
    wrong: 0,
    mistakes: [],
    timeLeft: timePerQuestion,
    legionPos: 5,
    firePos: 85,
    burned: false
  };

  showScreen("gameScreen");
  document.getElementById("totalQuestions").textContent = selected.length;
  updateChasePositions();
  nextQuestion();
}

function nextQuestion() {
  if (gameState.currentIndex >= gameState.questions.length) {
    endGame(true);
    return;
  }

  var q = gameState.questions[gameState.currentIndex];
  var askLatin = gameState.currentIndex % 2 === 0;

  var dirLabel = document.getElementById("directionLabel");
  var wordEl = document.getElementById("questionWord");

  if (askLatin) {
    dirLabel.textContent = "LATEIN \u2192 DEUTSCH";
    wordEl.textContent = q.en;
  } else {
    dirLabel.textContent = "DEUTSCH \u2192 LATEIN";
    wordEl.textContent = q.de;
  }

  wordEl.style.animation = "none";
  wordEl.offsetHeight;
  wordEl.style.animation = "";

  document.getElementById("questionNum").textContent = gameState.currentIndex + 1;
  document.getElementById("scoreDisplay").textContent = gameState.score;

  var correctAnswer = askLatin ? q.de : q.en;
  var wrongPool = [];
  for (var i = 0; i < vocab.length; i++) {
    if (vocab[i].en !== q.en) {
      wrongPool.push(askLatin ? vocab[i].de : vocab[i].en);
    }
  }

  var shuffledWrong = shuffle(wrongPool).slice(0, 3);
  var answers = shuffle(shuffledWrong.concat([correctAnswer]));

  var grid = document.getElementById("answersGrid");
  grid.innerHTML = "";
  for (var a = 0; a < answers.length; a++) {
    (function(answer) {
      var btn = document.createElement("button");
      btn.className = "answer-btn";
      btn.textContent = answer;
      btn.onclick = function() { handleAnswer(answer, correctAnswer, btn); };
      grid.appendChild(btn);
    })(answers[a]);
  }

  gameState.timeLeft = timePerQuestion;
  startTimer();
}

function startTimer() {
  clearInterval(timerInterval);
  var bar = document.getElementById("timerBar");
  var startTime = Date.now();
  var totalMs = timePerQuestion * 1000;

  bar.classList.remove("danger");

  timerInterval = setInterval(function() {
    var elapsed = Date.now() - startTime;
    var remaining = Math.max(0, 1 - elapsed / totalMs);
    bar.style.width = (remaining * 100) + "%";

    if (remaining < 0.25) { bar.classList.add("danger"); }
    else { bar.classList.remove("danger"); }

    if (remaining <= 0) {
      clearInterval(timerInterval);
      handleTimeout();
    }
  }, 50);
}

function handleAnswer(selected, correct, btn) {
  clearInterval(timerInterval);
  var buttons = document.querySelectorAll(".answer-btn");

  if (selected === correct) {
    btn.classList.add("correct");
    gameState.score += 10 + Math.floor(gameState.streak * 2);
    gameState.correct++;
    gameState.streak++;
    if (gameState.streak > gameState.maxStreak) gameState.maxStreak = gameState.streak;

    showFeedback(pickRandom(["RECTE!", "OPTIME!", "BENE!", "EUGE!", "VICTORIA!"]), "good");
    // Legion retreats from fire
    gameState.legionPos = Math.max(5, gameState.legionPos - 3);
  } else {
    btn.classList.add("wrong");
    for (var i = 0; i < buttons.length; i++) {
      if (buttons[i].textContent === correct) buttons[i].classList.add("correct");
    }
    gameState.wrong++;
    gameState.streak = 0;

    var q = gameState.questions[gameState.currentIndex];
    gameState.mistakes.push({ en: q.en, de: q.de });

    showFeedback(pickRandom(["FALSUM!", "MALE!", "CAVE!"]), "bad");
    // Legion marches closer to fire
    gameState.legionPos = Math.min(gameState.firePos - 8, gameState.legionPos + 10);
  }

  for (var j = 0; j < buttons.length; j++) { buttons[j].classList.add("disabled"); }
  updateStreakDisplay();
  updateChasePositions();
  checkBurned();

  if (!gameState.burned) {
    setTimeout(function() {
      gameState.currentIndex++;
      nextQuestion();
    }, 1200);
  }
}

function handleTimeout() {
  var q = gameState.questions[gameState.currentIndex];
  var askLatin = gameState.currentIndex % 2 === 0;
  var correct = askLatin ? q.de : q.en;

  var buttons = document.querySelectorAll(".answer-btn");
  for (var i = 0; i < buttons.length; i++) {
    if (buttons[i].textContent === correct) buttons[i].classList.add("correct");
    buttons[i].classList.add("disabled");
  }

  gameState.wrong++;
  gameState.streak = 0;
  gameState.mistakes.push({ en: q.en, de: q.de });

  showFeedback("\u23F0 ZU LANGSAM!", "bad");
  gameState.legionPos = Math.min(gameState.firePos - 8, gameState.legionPos + 14);
  updateStreakDisplay();
  updateChasePositions();
  checkBurned();

  if (!gameState.burned) {
    setTimeout(function() {
      gameState.currentIndex++;
      nextQuestion();
    }, 1500);
  }
}

function checkBurned() {
  if (gameState.legionPos >= gameState.firePos - 10) {
    gameState.burned = true;
    clearInterval(timerInterval);
    showBurnedAnimation();
  }
}

function showBurnedAnimation() {
  var overlay = document.createElement("div");
  overlay.className = "eaten-overlay";
  overlay.innerHTML = '<div class="eaten-scene">&#x1F525;&#x1F6E1;&#xFE0F;&#x1F525;</div><div class="eaten-text">IM FEUER GEFALLEN!</div>';
  document.body.appendChild(overlay);

  setTimeout(function() {
    overlay.remove();
    endGame(false);
  }, 2500);
}

function updateChasePositions() {
  if (!gameState) return;
  document.getElementById("legion").style.left = gameState.legionPos + "%";
}

function updateStreakDisplay() {
  var el = document.getElementById("streakDisplay");
  if (gameState.streak >= 2) {
    el.textContent = "\uD83D\uDD25 " + gameState.streak + "x Streak";
  } else {
    el.textContent = "";
  }
}

function showFeedback(text, type) {
  var el = document.createElement("div");
  el.className = "feedback " + type;
  el.textContent = text;
  document.body.appendChild(el);
  setTimeout(function() { el.remove(); }, 800);
}

function endGame(survived) {
  clearInterval(timerInterval);
  showScreen("resultScreen");

  var total = gameState.correct + gameState.wrong;
  var pct = total > 0 ? Math.round((gameState.correct / total) * 100) : 0;

  if (survived && pct >= 70) {
    document.getElementById("resultEmoji").textContent = "\uD83C\uDFC6";
    document.getElementById("resultTitle").textContent = "VICTORIA!";
    document.getElementById("resultTitle").className = "result-title win";
  } else if (survived) {
    document.getElementById("resultEmoji").textContent = "\uD83D\uDE05";
    document.getElementById("resultTitle").textContent = "KNAPP GERETTET!";
    document.getElementById("resultTitle").className = "result-title win";
  } else {
    document.getElementById("resultEmoji").textContent = "\uD83D\uDD25";
    document.getElementById("resultTitle").textContent = "NIEDERGANG!";
    document.getElementById("resultTitle").className = "result-title lose";
  }

  document.getElementById("resultStats").innerHTML =
    '<div class="stat-row"><span class="stat-label">Punkte</span><span class="stat-value">' + gameState.score + '</span></div>' +
    '<div class="stat-row"><span class="stat-label">Richtig</span><span class="stat-value">' + gameState.correct + ' / ' + total + '</span></div>' +
    '<div class="stat-row"><span class="stat-label">Quote</span><span class="stat-value">' + pct + '%</span></div>' +
    '<div class="stat-row"><span class="stat-label">Beste Serie</span><span class="stat-value">\uD83D\uDD25 ' + gameState.maxStreak + '</span></div>';

  var mistDiv = document.getElementById("mistakesList");
  if (gameState.mistakes.length > 0) {
    mistDiv.style.display = "block";
    var html = '<h3>&#x1F4D6; ZUM WIEDERHOLEN:</h3>';
    for (var i = 0; i < gameState.mistakes.length; i++) {
      html += '<div class="mistake-item">' + escapeHtml(gameState.mistakes[i].en) + ' \u2192 ' + escapeHtml(gameState.mistakes[i].de) + '</div>';
    }
    mistDiv.innerHTML = html;
  } else {
    mistDiv.style.display = "none";
  }
}

// ===== EDITOR =====
function showEditor() {
  showScreen("editorScreen");
  renderEditorRows();
}

function renderEditorRows() {
  var container = document.getElementById("vocabRows");
  container.innerHTML = "";
  for (var i = 0; i < vocab.length; i++) {
    (function(index) {
      var row = document.createElement("div");
      row.className = "vocab-row";
      row.innerHTML =
        '<input class="vocab-input" value="' + escapeHtml(vocab[index].en) + '" data-index="' + index + '" data-field="en" placeholder="Latein...">' +
        '<input class="vocab-input" value="' + escapeHtml(vocab[index].de) + '" data-index="' + index + '" data-field="de" placeholder="Deutsch...">' +
        '<button class="delete-btn" onclick="deleteRow(' + index + ')">&#x2715;</button>';
      container.appendChild(row);
    })(i);
  }
}

function addRow() {
  vocab.push({ en: "", de: "" });
  renderEditorRows();
  var wrap = document.getElementById("vocabTableWrap");
  wrap.scrollTop = wrap.scrollHeight;
  var inputs = wrap.querySelectorAll('.vocab-input[data-field="en"]');
  if (inputs.length) inputs[inputs.length - 1].focus();
}

function deleteRow(i) {
  vocab.splice(i, 1);
  renderEditorRows();
}

function saveAndBack() {
  var rows = document.querySelectorAll("#vocabRows .vocab-row");
  var newVocab = [];
  for (var i = 0; i < rows.length; i++) {
    var en = rows[i].querySelector('[data-field="en"]').value.trim();
    var de = rows[i].querySelector('[data-field="de"]').value.trim();
    if (en && de) newVocab.push({ en: en, de: de });
  }
  vocab = newVocab;
  saveVocab();
  showToast("\u2705 " + vocab.length + " Vokabeln gespeichert!");
  showScreen("startScreen");
}

function exportVocab() {
  var rows = document.querySelectorAll("#vocabRows .vocab-row");
  var exportData = [];
  for (var i = 0; i < rows.length; i++) {
    var en = rows[i].querySelector('[data-field="en"]').value.trim();
    var de = rows[i].querySelector('[data-field="de"]').value.trim();
    if (en && de) exportData.push({ en: en, de: de });
  }
  var blob = new Blob([JSON.stringify(exportData, null, 2)], { type: "application/json" });
  var a = document.createElement("a");
  a.href = URL.createObjectURL(blob);
  a.download = "latein_vokabeln.json";
  a.click();
  showToast("Vokabeln exportiert!");
}

function importVocab(event) {
  var file = event.target.files[0];
  if (!file) return;
  var reader = new FileReader();
  reader.onload = function(e) {
    try {
      var data = JSON.parse(e.target.result);
      if (Array.isArray(data) && data.length > 0 && data[0].en && data[0].de) {
        vocab = data;
        saveVocab();
        renderEditorRows();
        showToast(vocab.length + " Vokabeln importiert!");
      } else {
        showToast("Ung\u00fcltiges Format!");
      }
    } catch(err) {
      showToast("Datei konnte nicht gelesen werden!");
    }
  };
  reader.readAsText(file);
  event.target.value = "";
}

// ===== HELPERS =====
function pickRandom(arr) { return arr[Math.floor(Math.random() * arr.length)]; }

function escapeHtml(str) {
  return String(str).replace(/&/g, "&amp;").replace(/</g, "&lt;").replace(/>/g, "&gt;").replace(/"/g, "&quot;");
}

function showToast(msg) {
  var t = document.getElementById("toast");
  t.textContent = msg;
  t.classList.add("visible");
  setTimeout(function() { t.classList.remove("visible"); }, 2500);
}

// Keyboard support: 1-4 for answers
document.addEventListener("keydown", function(e) {
  if (!document.getElementById("gameScreen").classList.contains("active")) return;
  var key = parseInt(e.key, 10);
  if (key >= 1 && key <= 4) {
    var btns = document.querySelectorAll(".answer-btn:not(.disabled)");
    if (btns[key - 1]) btns[key - 1].click();
  }
});
